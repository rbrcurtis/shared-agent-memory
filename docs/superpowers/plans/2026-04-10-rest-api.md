```typescript
import { createHash, timingSafeEqual } from 'crypto';
import type { FastifyRequest, FastifyReply } from 'fastify';

export interface ApiKeyConfig {
  key: string;
  name: string;
  projects: string[] | null;  // null = full access
}

// Attached to request by auth hook
declare module 'fastify' {
  interface FastifyRequest {
    apiKey: ApiKeyConfig;
  }
}

export function parseApiKeys(envValue: string): ApiKeyConfig[] {
  let parsed: unknown;
  try {
    parsed = JSON.parse(envValue);
  } catch {
    throw new Error('API_KEYS env var is not valid JSON');
  }

  if (!Array.isArray(parsed)) {
    throw new Error('API_KEYS must be a JSON array');
  }

  return parsed.map((entry: unknown, i: number) => {
    if (typeof entry !== 'object' || entry === null) {
      throw new Error(`API_KEYS[${i}] is not an object`);
    }
    const obj = entry as Record<string, unknown>;
    if (typeof obj.key !== 'string' || !obj.key) {
      throw new Error(`API_KEYS[${i}].key is required`);
    }
    if (typeof obj.name !== 'string' || !obj.name) {
      throw new Error(`API_KEYS[${i}].name is required`);
    }
    if (obj.projects !== null && !Array.isArray(obj.projects)) {
      throw new Error(`API_KEYS[${i}].projects must be null or string array`);
    }
    return {
      key: obj.key as string,
      name: obj.name as string,
      projects: obj.projects as string[] | null,
    };
  });
}

function hashKey(key: string): Buffer {
  return createHash('sha256').update(key).digest();
}

export function validateKey(bearer: string, keys: ApiKeyConfig[]): ApiKeyConfig | null {
  if (!bearer) return null;
  const incomingHash = hashKey(bearer);

  for (const config of keys) {
    const configHash = hashKey(config.key);
    if (incomingHash.length === configHash.length && timingSafeEqual(incomingHash, configHash)) {
      return config;
    }
  }
  return null;
}

export function checkProjectAccess(key: ApiKeyConfig, project: string): boolean {
  if (project === '*') return true;  // wildcard always allowed, resolveProject handles scoping
  if (key.projects === null) return true;  // full access
  return key.projects.includes(project);
}

/**
 * Resolves the project parameter for storage queries.
 * - Specific project name → returns that string (used as exact filter)
 * - '*' with full-access key → returns undefined (no filter)
 * - '*' with restricted key → returns the key's project list
 */
export function resolveProject(
  key: ApiKeyConfig,
  project: string,
): string | string[] | undefined {
  if (project === '*') {
    return key.projects === null ? undefined : key.projects;
  }
  return project;
}

export function buildAuthHook(keys: ApiKeyConfig[]) {
  return async function authHook(request: FastifyRequest, reply: FastifyReply): Promise<void> {
    const authHeader = request.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      reply.code(401).send({ error: { code: 401, message: 'Missing or invalid Authorization header' } });
      return;
    }

    const token = authHeader.slice(7);
    const keyConfig = validateKey(token, keys);
    if (!keyConfig) {
      reply.code(401).send({ error: { code: 401, message: 'Invalid API key' } });
      return;
    }

    request.apiKey = keyConfig;
  };
}
```

```typescript
interface ConfigRouteDeps {
  qdrantUrl: string;
  collectionName: string;
  isModelReady: () => boolean;
}

export async function configRoutes(
  app: FastifyInstance,
  deps: ConfigRouteDeps,
): Promise<void> {
  // GET /health — no auth required, used by k8s probes
  app.get('/health', {
    schema: {
      tags: ['health'],
      summary: 'Liveness/readiness probe',
      response: { 200: healthResponse, 503: healthResponse },
    },
  }, async (_request, reply) => {
    const ready = deps.isModelReady();
    const status = ready ? 'ok' : 'starting';
    const code = ready ? 200 : 503;
    reply.code(code).send({ status, modelReady: ready });
  });

  // GET /api/v1/config — requires auth (registered under prefix)
  app.get('/api/v1/config', {
    schema: {
      tags: ['config'],
      summary: 'Server configuration and status',
      response: { 200: configResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async () => {
    return {
      data: {
        qdrantUrl: deps.qdrantUrl,
        collectionName: deps.collectionName,
        modelReady: deps.isModelReady(),
      },
    };
  });
}
```

```typescript
import type { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
import { StorageService } from '../../storage.js';
import { EmbeddingService } from '../../embeddings.js';
import { detectSecrets } from '../../secret-filter.js';
import { computeRetention, OVER_FETCH_MULTIPLIER, TOMBSTONE_THRESHOLD } from '../../retention.js';
import { checkProjectAccess, resolveProject } from '../middleware/auth.js';
import type { ApiKeyConfig } from '../middleware/auth.js';
import type { SearchResult } from '../../types.js';
import {
  storeMemoryBody, storeMemoryResponse,
  searchQuerystring, searchResponse,
  loadQuerystring, loadResponse,
  recentQuerystring, recentResponse,
  updateMemoryBody, memoryIdParams, successResponse,
  errorResponse,
} from '../schemas/memory.js';

interface MemoryRouteDeps {
  storage: StorageService;
  embeddings: EmbeddingService;
  log: (msg: string) => void;
}

function sendError(reply: FastifyReply, code: number, message: string): void {
  reply.code(code).send({ error: { code, message } });
}

export async function memoryRoutes(
  app: FastifyInstance,
  deps: MemoryRouteDeps,
): Promise<void> {
  const { storage, embeddings, log } = deps;

  // POST /api/v1/memories
  app.post('/api/v1/memories', {
    schema: {
      tags: ['memories'],
      summary: 'Store a new memory',
      body: storeMemoryBody,
      response: { 201: storeMemoryResponse, 400: errorResponse, 403: errorResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async (request: FastifyRequest<{ Body: { text: string; title: string; agent?: string; project: string; tags?: string[] } }>, reply) => {
    const { text, title, agent, project, tags } = request.body;
    const key = request.apiKey;

    if (!checkProjectAccess(key, project)) {
      sendError(reply, 403, `Access denied for project: ${project}`);
      return;
    }

    const detection = detectSecrets(text) || detectSecrets(title);
    if (detection) {
      sendError(reply, 400, `Memory rejected: detected secret (${detection.rule} at position ${detection.position}). Context: ${detection.snippet}`);
      return;
    }

    const resolved = resolveProject(key, project);
    // For store, * is not meaningful — require a specific project name
    if (resolved === undefined || Array.isArray(resolved)) {
      sendError(reply, 400, 'Cannot store with project=*. Specify a project name.');
      return;
    }

    const vector = await embeddings.generateEmbedding(text);
    const id = await storage.store({
      text,
      title,
      vector,
      agent: agent || 'unknown',
      project: resolved,
      tags: tags || [],
    });

    reply.code(201).send({ data: { id } });
  });

  // GET /api/v1/memories/search
  app.get('/api/v1/memories/search', {
    schema: {
      tags: ['memories'],
      summary: 'Search memories by semantic similarity',
      querystring: searchQuerystring,
      response: { 200: searchResponse, 400: errorResponse, 403: errorResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async (request: FastifyRequest<{ Querystring: { query: string; limit?: number; agent?: string; project: string; tags?: string } }>, reply) => {
    const { query, limit = 10, agent, project, tags } = request.query;
    const key = request.apiKey;

    if (!checkProjectAccess(key, project)) {
      sendError(reply, 403, `Access denied for project: ${project}`);
      return;
    }

    const resolved = resolveProject(key, project);
    const parsedTags = tags ? tags.split(',').map(t => t.trim()).filter(Boolean) : undefined;
    const vector = await embeddings.generateEmbedding(query);

    // If resolved is an array (restricted key with *), search each project and merge.
    // If resolved is string or undefined, single search.
    let allResults: SearchResult[] = [];

    if (Array.isArray(resolved)) {
      for (const proj of resolved) {
        const results = await storage.search({
          vector,
          queryText: query,
          limit: limit * OVER_FETCH_MULTIPLIER,
          agent,
          project: proj,
          tags: parsedTags,
        });
        allResults.push(...results);
      }
    } else {
      allResults = await storage.search({
        vector,
        queryText: query,
        limit: limit * OVER_FETCH_MULTIPLIER,
        agent,
        project: resolved,
        tags: parsedTags,
      });
    }

    // Retention pipeline
    const now = Date.now();
    const toTombstone: string[] = [];
    const scored: Array<{ id: string; title: string; score: number }> = [];

    for (const r of allResults) {
      const lastAccessed = r.last_accessed || r.created_at;
      const daysSince = (now - new Date(lastAccessed).getTime()) / (1000 * 60 * 60 * 24);
      const stability = r.stability ?? 1.0;
      const retention = computeRetention(daysSince, stability);

      if (retention < TOMBSTONE_THRESHOLD) {
        toTombstone.push(r.id);
      } else {
        scored.push({ id: r.id, title: r.title, score: r.score * retention });
      }
    }

    scored.sort((a, b) => b.score - a.score);
    const final = scored.slice(0, limit);

    // Fire-and-forget tombstoning
    if (toTombstone.length > 0) {
      storage.tombstoneMemories(toTombstone).catch((err: unknown) => {
        log(`Failed to tombstone memories: ${err}`);
      });
    }

    return { data: final };
  });

  // GET /api/v1/memories/load
  app.get('/api/v1/memories/load', {
    schema: {
      tags: ['memories'],
      summary: 'Load full text of memories by ID',
      querystring: loadQuerystring,
      response: { 200: loadResponse, 400: errorResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async (request: FastifyRequest<{ Querystring: { ids: string; project: string } }>, reply) => {
    const { ids: idsStr, project } = request.query;
    const key = request.apiKey;

    if (!checkProjectAccess(key, project)) {
      sendError(reply, 403, `Access denied for project: ${project}`);
      return;
    }

    const ids = idsStr.split(',').map(id => id.trim()).filter(Boolean);
    if (ids.length === 0) {
      sendError(reply, 400, 'ids parameter is required');
      return;
    }

    const results = await storage.getByIds(ids);

    // Filter by project access
    const filtered = results.filter(r => {
      if (key.projects === null) return true;  // full access
      return key.projects.includes(r.project);
    });

    // Reinforce loaded memories (fire and forget)
    if (filtered.length > 0) {
      storage.reinforceMemories(
        filtered.map(r => ({ id: r.id, accessCount: r.access_count ?? 0 }))
      ).catch((err: unknown) => {
        log(`Failed to reinforce memories: ${err}`);
      });
    }

    const data = filtered.map(r => ({
      id: r.id,
      title: r.title,
      text: r.text,
      agent: r.agent,
      project: r.project,
      tags: r.tags,
      created_at: r.created_at,
      last_accessed: r.last_accessed,
      access_count: r.access_count,
    }));

    return { data };
  });

  // GET /api/v1/memories/recent
  app.get('/api/v1/memories/recent', {
    schema: {
      tags: ['memories'],
      summary: 'List recently created memories',
      querystring: recentQuerystring,
      response: { 200: recentResponse, 403: errorResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async (request: FastifyRequest<{ Querystring: { limit?: number; days?: number; project: string } }>, reply) => {
    const { limit = 10, days = 30, project } = request.query;
    const key = request.apiKey;

    if (!checkProjectAccess(key, project)) {
      sendError(reply, 403, `Access denied for project: ${project}`);
      return;
    }

    const resolved = resolveProject(key, project);

    let allResults: SearchResult[] = [];

    if (Array.isArray(resolved)) {
      for (const proj of resolved) {
        const results = await storage.listRecent(limit, days, proj);
        allResults.push(...results);
      }
      allResults.sort((a, b) => new Date(b.created_at).getTime() - new Date(a.created_at).getTime());
      allResults = allResults.slice(0, limit);
    } else {
      allResults = await storage.listRecent(limit, days, resolved);
    }

    const data = allResults.map(r => ({
      id: r.id,
      title: r.title,
      created_at: r.created_at,
    }));

    return { data };
  });

  // PUT /api/v1/memories/:id
  app.put<{ Params: { id: string }; Body: { text: string; title?: string } }>('/api/v1/memories/:id', {
    schema: {
      tags: ['memories'],
      summary: 'Update an existing memory',
      params: memoryIdParams,
      body: updateMemoryBody,
      response: { 200: successResponse, 400: errorResponse, 403: errorResponse, 404: errorResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async (request, reply) => {
    const { id } = request.params;
    const { text, title: newTitle } = request.body;
    const key = request.apiKey;

    // Fetch-then-check: verify existing memory's project is in key's allowed list
    const existing = await storage.getByIds([id]);
    if (existing.length === 0) {
      sendError(reply, 404, `Memory not found: ${id}`);
      return;
    }

    if (!checkProjectAccess(key, existing[0].project)) {
      sendError(reply, 403, `Access denied for project: ${existing[0].project}`);
      return;
    }

    const title = newTitle || existing[0].title || '';

    const detection = detectSecrets(text) || detectSecrets(title);
    if (detection) {
      sendError(reply, 400, `Memory rejected: detected secret (${detection.rule} at position ${detection.position}). Context: ${detection.snippet}`);
      return;
    }

    const vector = await embeddings.generateEmbedding(text);
    await storage.update(id, {
      text,
      title,
      vector,
      agent: existing[0].agent || 'unknown',
      project: existing[0].project,
      tags: existing[0].tags || [],
    });

    return { data: { success: true } };
  });

  // DELETE /api/v1/memories/:id
  app.delete<{ Params: { id: string } }>('/api/v1/memories/:id', {
    schema: {
      tags: ['memories'],
      summary: 'Delete a memory',
      params: memoryIdParams,
      response: { 200: successResponse, 403: errorResponse, 404: errorResponse },
      security: [{ bearerAuth: [] }],
    },
  }, async (request, reply) => {
    const { id } = request.params;
    const key = request.apiKey;

    // Fetch-then-check
    const existing = await storage.getByIds([id]);
    if (existing.length === 0) {
      sendError(reply, 404, `Memory not found: ${id}`);
      return;
    }

    if (!checkProjectAccess(key, existing[0].project)) {
      sendError(reply, 403, `Access denied for project: ${existing[0].project}`);
      return;
    }

    await storage.delete(id);
    return { data: { success: true } };
  });
}
```

```typescript
import Fastify from 'fastify';
import cors from '@fastify/cors';
import swagger from '@fastify/swagger';
import swaggerUi from '@fastify/swagger-ui';
import { StorageService } from '../storage.js';
import { EmbeddingService } from '../embeddings.js';
import { ServerConfig } from '../types.js';
import { parseApiKeys, buildAuthHook } from './middleware/auth.js';
import { memoryRoutes } from './routes/memories.js';
import { configRoutes } from './routes/config.js';

export interface AppConfig {
  qdrantUrl: string;
  qdrantApiKey?: string;
  collectionName: string;
  apiKeysJson: string;
  port: number;
}

function log(msg: string): void {
  const timestamp = new Date().toISOString();
  console.error(`[${timestamp}] ${msg}`);
}

export async function buildApp(config: AppConfig) {
  const app = Fastify({ logger: false });

  // Parse API keys
  const apiKeys = parseApiKeys(config.apiKeysJson);
  log(`Loaded ${apiKeys.length} API key(s)`);

  // Swagger
  await app.register(swagger, {
    openapi: {
      info: {
        title: 'Shared Memory API',
        description: 'REST API for shared AI agent memory backed by Qdrant',
        version: '1.0.0',
      },
      components: {
        securitySchemes: {
          bearerAuth: {
            type: 'http',
            scheme: 'bearer',
            description: 'API key prefixed with sm_',
          },
        },
      },
    },
  });

  await app.register(swaggerUi, {
    routePrefix: '/docs',
  });

  // CORS
  await app.register(cors);

  // Auth hook — skip for /health and /docs
  const authHook = buildAuthHook(apiKeys);
  app.addHook('onRequest', async (request, reply) => {
    const url = request.url;
    if (url === '/health' || url.startsWith('/docs')) return;
    await authHook(request, reply);
  });

  // Global error handler
  app.setErrorHandler((error, _request, reply) => {
    const statusCode = error.statusCode || 500;
    log(`Error: ${error.message}`);
    reply.code(statusCode).send({
      error: {
        code: statusCode,
        message: error.message,
      },
    });
  });

  // Initialize storage
  const storageConfig: ServerConfig = {
    qdrantUrl: config.qdrantUrl,
    qdrantApiKey: config.qdrantApiKey,
    collectionName: config.collectionName,
    defaultAgent: 'unknown',
    defaultProject: 'default',
  };
  const storage = new StorageService(storageConfig);
  await storage.initialize();
  log(`Connected to Qdrant at ${config.qdrantUrl}`);

  // Initialize embeddings
  const embeddings = EmbeddingService.getInstance();
  log('Loading embedding model...');
  await embeddings.initialize();
  let modelReady = true;
  log('Embedding model ready');

  // Register routes
  await configRoutes(app, {
    qdrantUrl: config.qdrantUrl,
    collectionName: config.collectionName,
    isModelReady: () => modelReady,
  });

  await memoryRoutes(app, { storage, embeddings, log });

  return app;
}

export async function startServer(): Promise<void> {
  const config: AppConfig = {
    qdrantUrl: process.env.QDRANT_URL || '',
    qdrantApiKey: process.env.QDRANT_API_KEY,
    collectionName: process.env.COLLECTION_NAME || 'shared_agent_memory',
    apiKeysJson: process.env.API_KEYS || '[]',
    port: parseInt(process.env.PORT || '3000', 10),
  };

  if (!config.qdrantUrl) {
    console.error('QDRANT_URL environment variable is required');
    process.exit(1);
  }

  if (!config.apiKeysJson || config.apiKeysJson === '[]') {
    console.error('WARNING: No API keys configured. Set API_KEYS env var.');
  }

  const app = await buildApp(config);

  await app.listen({ port: config.port, host: '0.0.0.0' });
  log(`API server listening on port ${config.port}`);
  log(`Swagger docs at http://0.0.0.0:${config.port}/docs`);
}

// Run if this file is the entry point
const isMain = process.argv[1]?.endsWith('api/server.js') ||
  process.argv[1]?.endsWith('api/server.ts');
if (isMain) {
  startServer().catch((err) => {
    console.error('Fatal error:', err);
    process.exit(1);
  });
}
```
